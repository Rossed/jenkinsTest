pipeline {
    agent any

    parameters {
            string(name: 'Username', defaultValue: 'Ross', description: 'Who is running the build')
        }
    stages {    
        stage('Build') {
            steps {
                echo "Hello ${params.Username}, we are building..."
                sh 'mvn clean install assembly:single'
                echo "Build Success"
            }
        }
        stage('Test') {
            parallel linux: {
                node('linux') {
                    echo 'test 1 running'
                }
            },
            windows: {
                node('windows') {
                    echo 'test 2 running'
                }
            }
        }
        stage('Deploy') {
            steps {
                echo 'Deploying....'
            }
        }
    }
}
