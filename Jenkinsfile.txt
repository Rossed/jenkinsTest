pipeline {
    agent any

    parameters {
            string(name: 'Username', defaultValue: 'Ross', description: 'Who is running the build')
        }
    stages {    
        stage('Build') {
            steps {
                echo "Hello ${params.Username}, we are building..."
                sh 'mvn clean install assembly:single'
                echo "Build Success"
            }
        }
        stage('Test') {
            parallel test1:
            node('test1') {
                echo 'test1 is running...'
            }, 
            test2:
            node('test2') {
                echo 'test 2 is running'
            }
        }
        stage('Deploy') {
            steps {
                echo 'Deploying....'
            }
        }
    }
}
