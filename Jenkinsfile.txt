pipeline {
    agent any

    parameters {
            string(name: 'Username', defaultValue: 'Ross', description: 'How should I greet the world?')
        }
    stages {    
        stage('Build') {
            steps {
                echo "Hello ${params.Username}, we are building..."
                sh 'mvn clean install assembly:single'
                echo "Build Success"
            }
        }
        stage('Test') {
            steps {
                echo 'Testing..'
            }
        }
        stage('Deploy') {
            steps {
                echo 'Deploying....'
            }
        }
    }
}
